// Debian 11 (bullseye) mit debootstrap installieren (nicht die minimale Version)

$ sudo debootstrap --arch=amd64 bullseye /mnt 

// Falls der "Release signed by unknown key" Fehler auftritt, kann der "--no-check-gpg" Flag verwendet werden.
// sudo debootstrap --arch=amd64 --no-check-gpg bullseye /mnt 

// F端r chroot notwendige Verzeichnisse einbinden:
$ sudo mount -o bind /dev /mnt/dev
$ sudo mount -o bind /dev/pts /mnt/dev/pts
$ sudo mount -t sysfs /sys /mnt/sys
$ sudo mount -t proc /proc /mnt/proc
$ sudo cp /proc/mounts /mnt/etc/mtab  

// F端r die Internetverbindung resolv.conf kopieren:
$ sudo cp /etc/resolv.conf /mnt/etc/resolv.conf 

// Mit chroot in das installierte System wechseln:
$ sudo chroot /mnt
